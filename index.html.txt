        function renderCoordinates() {
            const container = document.getElementById('coordinates-container');
            container.innerHTML = coordinatesData.map(c => `
                <div class="bg-slate-900/60 border border-slate-800/80 p-6 rounded-2xl shadow-xl">

        function initializeAndRenderWorkbook() {
            if (!localStorage.getItem('matrix_workbook_cache')) {
                localStorage.setItem('matrix_workbook_cache', JSON.stringify(initialWorkbookData));
            }
            renderWorkbook();
        }

        function renderWorkbook() {
            const data = JSON.getObj('matrix_workbook_cache');
            const container = document.getElementById('workbook-container');
            
            container.innerHTML = data.map(e => `
                <div class="bg-slate-950 border border-slate-800 p-5 rounded-2xl shadow-inner space-y-4">
                    <div class="flex justify-between items-center border-b border-slate-900 pb-3">
                        <span class="text-xs font-mono font-bold text-amber-400 tracking-wider bg-amber-950/20 border border-amber-900/40 px-2 py-1 rounded">${e.section} — ${e.title}</span>
                        <label class="inline-flex items-center text-xs text-slate-400 cursor-pointer select-none">
                            <input type="checkbox" id="check-${e.id}" onchange="syncWorkbookState('${e.id}')" ${e.checked ? 'checked' : ''} class="rounded bg-slate-900 border-slate-700 text-blue-600 focus:ring-0 mr-2 h-4 w-4"> 
                            Declare Difference
                        </label>
                    </div>
                    <textarea id="text-${e.id}" oninput="syncWorkbookState('${e.id}')" rows="3" class="w-full bg-slate-900 border border-slate-800 rounded-xl p-3 text-xs text-slate-300 font-mono focus:outline-none focus:border-blue-600 transition leading-relaxed" placeholder="Frame exception parameters here using standard constitutional language...">${e.text}</textarea>
                </div>
            `).join('');
        }

        function syncWorkbookState(id) {
            const data = JSON.getObj('matrix_workbook_cache');
            const item = data.find(e => e.id === id);
            if (item) {
                item.checked = document.getElementById(`check-${id}`).checked;
                item.text = document.getElementById(`text-${id}`).value;
                localStorage.setItem('matrix_workbook_cache', JSON.stringify(data));
            }
        }

        JSON.getObj = function(key) {
            return JSON.parse(localStorage.getItem(key));
        }

        // Live Clock Engine Execution Logic
        function triggerClock(category, prompt, template) {
            document.getElementById('active-drill-placeholder').classList.add('hidden');
            document.getElementById('active-drill-zone').classList.remove('hidden');
            document.getElementById('shield-reveal-zone').classList.add('hidden');
            document.getElementById('candidate-live-entry').value = '';
            document.getElementById('active-prompt-text').innerText = category + ": " + prompt;
            document.getElementById('active-template-text').innerText = template;
            
            let durationRemaining = 60;
            const clockDisplay = document.getElementById('clock-countdown');
            clockDisplay.innerText = durationRemaining;
            clockDisplay.className = "text-3xl font-mono font-black text-emerald-400 tracking-tighter";
            
            clearInterval(clockTicker);
            clockTicker = setInterval(() => {
                durationRemaining--;
                clockDisplay.innerText = durationRemaining;
                if(durationRemaining <= 15) {
                    clockDisplay.className = "text-3xl font-mono font-black text-red-500 tracking-tighter animate-pulse";
                } else if(durationRemaining <= 30) {
                    clockDisplay.className = "text-3xl font-mono font-black text-amber-500 tracking-tighter";
                }
                if(durationRemaining <= 0) {
                    clearInterval(clockTicker);
                    clockDisplay.innerText = "STOP TALKING";
                }
            }, 1000);
        }

        function revealShieldText() {
            document.getElementById('shield-reveal-zone').classList.remove('hidden');
            clearInterval(clockTicker);
        }
